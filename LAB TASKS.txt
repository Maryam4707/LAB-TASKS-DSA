#include <iostream>
using namespace std;
struct Node {
    int data;
    Node* next;
};

void insertEnd(Node*& head, int value)
 {
    Node* newNode = new Node;
    newNode->data = value;
    newNode->next = NULL;
    if (head == NULL) 
	{
        head = newNode;
        return;
    }
    Node* temp = head;
    while (temp->next != NULL)
	 {
        temp = temp->next;
    }
    temp->next = newNode;
}
void display(Node* head) 
{
    Node* temp = head;
    while (temp != NULL)
	 {
        cout << temp->data << " -> ";
        temp = temp->next;
    }
    cout << "NULL:"<<endl;
}
int main() 
{
    Node* head = NULL; 
    insertEnd(head, 10);
    insertEnd(head, 20);
    insertEnd(head, 30);
    insertEnd(head, 40);
    cout << "Linked List: ";
    display(head);
    return 0;
}

 {
    Node* newNode = new Node{value, NULL, NULL};
    if (!head)
	 { 
        head = newNode;
        return;
    }
    Node* temp = head;
    while (temp->next) temp = temp->next; 
    temp->next = newNode;
    newNode->prev = temp;
}
void deleteLast(Node*& head) 
{
    if (!head) 
	{
        cout << "List is empty!";
        return;
    }
    if (!head->next) 
	{ 
        delete head;
        head = NULL;
        return;
    }
    Node* temp = head;
    while (temp->next) temp = temp->next; 
    temp->prev->next = NULL;
    delete temp;
}

        head = head->next;
